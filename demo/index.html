<script>   
<button class="select-enable">click to enable</button>   
<button class="select-disabled">click to disable</button>               
    $(document).ready(function() {
        $('.select-disable').asSelect({
            skin: 'simple',
        }); 
        
        $('.select-enable').on('click', function() {
            // enable
            $('.select-disable').asSelect('enable'); 
        });
        $('.select-disabled').on('click', function() {
            // here call disable method
            $('.select-disable').asSelect('disable'); 
        });  
    });
</script>

build use load function


<div class="build-use-load"></div>
<script>                   
    $(document).ready(function() {
        $('.build-use-load').asSelect({
            valueField: 'url',
            labelField: 'name',
            preload: true,
            load: function(callback){
                $.ajax({
                    url: 'https://api.github.com/users/amazingSurge/repos',
                    type: 'GET',
                    error: function() {
                        callback();
                    },
                    success: function(res) {
                        var data = [];
                        $.each(res.slice(0, 10), function(i, item){
                            data.push({
                                value: item.url,
                                text: item.name,
                                fork: item.fork,
                                id: item.id
                            });
                        });
                        callback(data);
                    }
                });
            },
            render: {
                option: function(item){
                    return '<span class="name"><i class="icon ' + (item.fork ? 'fork' : 'source') + '"></i>' + item.text + '</span>' +
                            '<span class="id">' + item.id + '</span>';
                }
            }
        });
    });
</script>
